    # AsyncBulkJob

    Позволяет удобно работать с параллельными асинхронными операциями с возможностью отслеживания прогресса выполнения операций.

    # Пример

    import { AsyncBulkJob, Result } from '@cse/core';

function createSaveProductsJob(products) {
  const job = new AsyncBulkJob(saveProduct, products);

  job.interceptors.done.use(({ data, errors }) => ({
    status: Object.keys(errors).length ? 'есть ошибки' : 'нет ошибок',
    errors: errors,
    data,
  }));

  return job;
}

async function saveProduct(product) {
  // ...
  return Result.ok(data);
}

// Использование

function saveProducts(products) {
  // создаем job
  const job = createSaveProductsJob(products);

  // подписываемся на события и обрабатываем их
  job.on('start', () => {
    console.log('Начало сохранения');
  });
  job.on('progress', ({ total, success, errors, percentage }) => {
    console.log('Прогресс сохранения: ' + percentage);
    console.log('Всего: ' + total);
    console.log('Сохранено: ' + success);
    console.log('Ошибок: ' + errors);
  });
  job.on('done', ({ status, data, errors }) => {
    console.log('Готово');
    console.log(status, data, errors);
  });

  // запускаем
  job.execute();
}
