    # Result

    Контейнер результата. Основное применение - разделение ожидаемых ошибок поведения от исключений (например: ошибка валидации/синтаксическая ошибка (исключение))

    # Как использовать

    import { Result } from '@cse/core';

async function saveProduct(product) {
  try {
    const response = await api.create('products', product);
    return Result.ok('товар сохранен');
  } catch (error) {
    // если ошибка HTTP, тогда обрабатываем ее, иначе пробрасываем
    if (error.response) {
      return Result.error(new Error('Не удалось сохранить товар'));
    }
    throw error;
  }
}

// Клиентский код будет выглядеть следующим образом

const result = await saveProduct({
  id: 2,
  category: 'leather_goods',
  name: 'Billy',
});

// Обработка результата похожа на работу с Promise
result
  .ok((message) => {
    notify(message);
  })
  .error((error) => {
    showError(error.message);
  });

// Все исключения можем обработать отдельно
window.onerror = function (error) {
  loggerService.logError(error.message);
};

    По аналогии с Promise, Result можно составлять в цепочки

    async function sendMessage(message) {
  const result = await sendMessageViaEmail(message);

  return result
    .ok((data) => {
      return { status: 'success' };
    })
    .error((error) => {
      return new SendMessageClientError('Ошибка');
    });
}
