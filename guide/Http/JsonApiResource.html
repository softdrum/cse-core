    # JsonApiResource

    # Описание

    JsonApiResource - обертка над JsonApiClient, для удобного создания ресурсов для чтения данных в формате JsonApi.

    TIP

    Каждый метод JsonApiResource возвращает контейнер Result

    # Пример

    import { createJsonApiResource, JsonApiClient } from '@cse/core';

const productsResource = createJsonApiResource('products', {
  jsonApiClient: new JsonApiClient(), // инстанс JsonApiClient
});

async function getProducts() {
  const result = await productsResource.findAll({
    page: {
      number: 1,
      size: 42,
    },
    filter: {
      type: 'electrical_instruments',
    },
    sort: 'name,-price',
    include: 'product_images',
  });

  return result
    .ok((productDTOs) =>
      productDTOs.map((dto) => {
        return {
          id: dto.id,
          type: dto.type,
          name: dto.name + dto.category,
        };
      })
    )
    .error(() => new Error('Не удалось загрузить товары'));
}

    import { JsonApiResource, JsonApiClient } from '@cse/core';

class ProductsResource extends JsonApiResource {
  constructor(settings) {
    super('products', settings)
  },

  // При необходимости есть возможность переопределения и добавление методов JsonApiResource
  findAll(params) {
    try {
      const response = await this._jsonApiClient.find(this._resource, id);

      const normalized = response.data.map(dto => {
         return {
            id: dto.id,
            type: dto.type,
            name: dto.name + dto.category,
          };
      })

      return Result.ok({
        ...response,
        data: normalized,
      });
    } catch (error) {
      if (error instanceof HttpError) {
        return Result.error(new Error('Не удалось загрузить товары'));
      }
      throw error;
    }
  }
}

const productsResource = new ProductsResource({ jsonApiClient: new JsonApiClient('url') });

async function getProducts() {
  const result = await productsResource.findAll({
    page: {
      number: 1,
      size: 42,
    },
    filter: {
      type: 'electrical_instruments',
    },
    sort: 'name,-price',
    include: 'product_images',
  });

  return result
}
