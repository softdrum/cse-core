    # JsonApiClient

    # Описание

    JsonApiClient - клиент для работы с REST-API, работающего по спецификации JSON API Внутри он проводит десереализацию данных, подставляя included и relationships.

    # Response

    {
  "type": "users",
  "id": "1",
  "attributes": {
    "first-name": "Sandro",
    "last-name": "Munda"
  }
}

    # Десереализованные данные

    {
  "id": "1",
  "first-name": "Sandro",
  "last-name": "Munda"
}

    # Пример

    import { JsonApiClient } from '@cse/core';

const api = new JsonApiClient({ baseUrl: 'http://jsonapi.address.com/api/v1' });

const RESOURCE_NAME = 'products';

async function getProducts() {
  // http://jsonapi.address.com/api/v1/products
  const response = await api.findAll(RESOURCE_NAME, {
    // ?page[number]=1&page[size]=42
    page: {
      number: 1,
      size: 42,
    },
    // &filter[type]=electrical_instruments
    filter: {
      type: 'electrical_instruments',
    },
    // &sort=name,-price
    sort: 'name,-price',
    // &include=product_images
    include: 'product_images',
  });
  return response;
}

function getProductById(id) {
  return api.find(RESOURCE_NAME, id);
}

function createProduct(product) {
  return api.create(RESOURCE_NAME, product);
}

function updateProduct(product) {
  return api.update(RESOURCE_NAME, product);
}

function deleteProduct(id) {
  return api.delete(RESOURCE_NAME, id);
}
